myvariable dd 8 ; put uint 8 in memory right here
 
main:
mov ecx, 10

loop:
cmp ecx, 0
je  end
mov eax, [myvariable]
mov eax, [myvariable] ; copy myvariable memory to register A
add dword[myvariable], eax ; add myvariable by itself
dec ecx               ; repeat  it 10 times
jmp loop

end:
jmp end
